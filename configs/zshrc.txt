# Git branch in prompt.
parse_git_branch() {
   behind=`git status uno 2> /dev/null | grep behind`
   if [ -z "$behind" ]; then
      git branch 2> /dev/null | sed -e '/^[^*]/d' -e 's/* \(.*\)/ (\1)/'
   else
      git branch 2> /dev/null | sed -e '/^[^*]/d' -e 's/* \(.*\)/ (\1 â†‘)/'
   fi
}
